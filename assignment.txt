void main() {
  // Qno1
  // Three types of operators
  //1)Arithmetic    2)Equality and Relational Operator  3)Logical Operator
  
  // 1) Arithmetic
  var a=10;
  var b=5;
  
  num add =a+b;
  int sub=a-b;
  var mul=a*b;
  num div=a/b;
  num mod=a%b;
  num inc=++a;
  num dec=--b;
  
  print("${add}\n${sub}\n${mul}\n${div}\n${mod}\n${inc}\n${dec}");
  
  //2)Equality and Relational Operator
  var greater = a>b;
  var less = b<a;
  var grequal = a>=b;
  var lessequal= a<=b;
  var equal = a==b;
  var not = a!=b;
  print("${greater}\n${less}\n${grequal}\n${lessequal}\n${equal}\n${not}");
  
  //3)Logical Operator
  var c=20;
  var d=30;
  var and = (a>b) && (c>d);
  var or= a>b || c>d;
  var no= !(a>b);
  print("${and}\n${or}\n${no}");
}
